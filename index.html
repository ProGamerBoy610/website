// Initialize the application function init() { checkAuthentication(); if (isAuthenticated) { loadSavedData(); setupNavigation(); setupEventListeners(); updateAllDisplays(); } else { setupLoginListeners(); } } // Authentication functions function checkAuthentication() { const savedAuth = localStorage.getItem('dashboard_auth'); const savedTimestamp = localStorage.getItem('dashboard_auth_timestamp'); const currentTime = Date.now(); // Check if authentication is valid (expires after 24 hours) if (savedAuth === 'true' && savedTimestamp && (currentTime - parseInt(savedTimestamp)) < 86400000) { isAuthenticated = true; showDashboard(); } else { // Clear expired auth localStorage.removeItem('dashboard_auth'); localStorage.removeItem('dashboard_auth_timestamp'); showLogin(); } } function setupLoginListeners() { const keyInput = document.getElementById('keyInput'); keyInput.addEventListener('keypress', function(e) { if (e.key === 'Enter') { authenticate(); } }); keyInput.addEventListener('input', function() { this.classList.remove('error'); document.getElementById('errorMessage').classList.remove('show'); }); } function authenticate() { const enteredKey = document.getElementById('keyInput').value; const keyInput = document.getElementById('keyInput'); const errorMessage = document.getElementById('errorMessage'); if (enteredKey === ACCESS_KEY) { isAuthenticated = true; localStorage.setItem('dashboard_auth', 'true'); localStorage.setItem('dashboard_auth_timestamp', Date.now().toString()); showDashboard(); loadSavedData(); setupNavigation(); setupEventListeners(); updateAllDisplays(); showNotification('Welcome to your dashboard!', 'success'); } else { keyInput.classList.add('error'); errorMessage.classList.add('show'); keyInput.value = ''; keyInput.focus(); } } function logout() { if (confirm('Are you sure you want to logout?')) { isAuthenticated = false; localStorage.removeItem('dashboard_auth'); localStorage.removeItem('dashboard_auth_timestamp'); showLogin(); // Reset all data currentTab = 'dashboard'; document.querySelectorAll('.nav-item').forEach(item => { item.classList.remove('active'); }); document.querySelector('[data-tab="dashboard"]').classList.add('active'); document.querySelectorAll('.tab-content').forEach(content => { content.classList.remove('active'); }); document.getElementById('dashboard').classList.add('active'); } } function showLogin() { document.getElementById('loginContainer').style.display = 'flex'; document.getElementById('dashboardContainer').classList.remove('authenticated'); document.getElementById('keyInput').focus(); } function showDashboard() { document.getElementById('loginContainer').style.display = 'none'; document.getElementById('dashboardContainer').classList.add('authenticated'); } // Data persistence functions function saveData() { if (!isAuthenticated) return; const data = { currentBalance, transactions, links, todos, images: images.map(img => ({ ...img, // Keep only essential data for images to avoid localStorage size limits src: img.src.length > 50000 ? '[Large Image Data]' : img.src })), lastSaved: Date.now() }; try { localStorage.setItem('dashboard_data', JSON.stringify(data)); showSaveStatus(); } catch (error) { console.error('Error saving data:', error); showNotification('Error saving data - storage might be full', 'error'); } } function loadSavedData() { if (!isAuthenticated) return; try { const savedData = localStorage.getItem('dashboard_data'); if (savedData) { const data = JSON.parse(savedData); currentBalance = data.currentBalance || 0; transactions = data.transactions || []; links = data.links || []; todos = data.todos || []; images = data.images || []; // Convert date strings back to Date objects transactions.forEach(t => { if (t.date) t.date = new Date(t.date); }); todos.forEach(t => { if (t.createdAt) t.createdAt = new Date(t.createdAt); if (t.dueDate) t.dueDate = new Date(t.dueDate); }); links.forEach(l => { if (l.dateAdded) l.dateAdded = new Date(l.dateAdded); }); images.forEach(i => { if (i.uploadDate) i.uploadDate = new Date(i.uploadDate); }); console.log('Data loaded successfully'); } } catch (error) { console.error('Error loading data:', error); showNotification('Error loading saved data', 'error'); } } function showSaveStatus() { const saveStatus = document.getElementById('saveStatus'); saveStatus.classList.add('show'); setTimeout(() => { saveStatus.classList.remove('show'); }, 2000); } // Navigation setup function setupNavigation() { const navItems = document.querySelectorAll('.nav-item'); navItems.forEach(item => { item.addEventListener('click', function() { const tabId = this.getAttribute('data-tab'); switchTab(tabId); }); }); } function switchTab(tabId) { // Update navigation document.querySelectorAll('.nav-item').forEach(item => { item.classList.remove('active'); }); document.querySelector(`[data-tab="${tabId}"]`).classList.add('active'); // Update content document.querySelectorAll('.tab-content').forEach(content => { content.classList.remove('active'); }); document.getElementById(tabId).classList.add('active'); currentTab = tabId; } // Event listeners setup function setupEventListeners() { // Period selector buttons document.querySelectorAll('.period-btn').forEach(btn => { btn.addEventListener('click', function() { document.querySelectorAll('.period-btn').forEach(b => b.classList.remove('active')); this.classList.add('active'); updateRevenueStats(); }); }); // Enter key handlers document.getElementById('revenueAmount').addEventListener('keypress', function(e) { if (e.key === 'Enter') addRevenue(); }); document.getElementById('expenseAmount').addEventListener('keypress', function(e) { if (e.key === 'Enter') addExpense(); }); document.getElementById('linkUrl').addEventListener('keypress', function(e) { if (e.key === 'Enter') addLink(); }); document.getElementById('todoTitle').addEventListener('keypress', function(e) { if (e.key === 'Enter') addTodo(); }); // Image upload setup const uploadArea = document.getElementById('uploadArea'); const imageUpload = document.getElementById('imageUpload'); uploadArea.addEventListener('click', () => imageUpload.click()); uploadArea.addEventListener('dragover', handleDragOver); uploadArea.addEventListener('drop', handleDrop); uploadArea.addEventListener('dragleave', handleDragLeave); imageUpload.addEventListener('change', handleFileSelect); // Modal setup document.getElementById('imageModal').addEventListener('click', closeModal); } // Revenue functions function addRevenue() { if (!isAuthenticated) return; const amount = parseFloat(document.getElementById('revenueAmount').value); const description = document.getElementById('revenueDescription').value; const category = document.getElementById('revenueCategory').value; if (!amount || amount <= 0) { showNotification('Please enter a valid amount', 'error'); return; } const transaction = { id: Date.now(), type: 'income', amount: amount, description: description || 'Revenue', category: category, date: new Date(), timestamp: Date.now() }; transactions.unshift(transaction); currentBalance += amount; // Clear inputs document.getElementById('revenueAmount').value = ''; document.getElementById('revenueDescription').value = ''; updateAllDisplays(); saveData(); showNotification(`Added ${amount.toFixed(2)} revenue!`, 'success'); } function addExpense() { if (!isAuthenticated) return; const amount = parseFloat(document.getElementById('expenseAmount').value); const description = document.getElementById('expenseDescription').value; const category = document.getElementById('expenseCategory').value; if (!amount || amount <= 0) { showNotification('Please enter a valid amount', 'error'); return; } if (amount > currentBalance) { if (!confirm(`This expense (${amount.toFixed(2)}) exceeds your current balance (${currentBalance.toFixed(2)}). Continue?`)) { return; } } const transaction = { id: Date.now(), type: 'expense', amount: amount, description: description || 'Expense', category: category, date: new Date(), timestamp: Date.now() }; transactions.unshift(transaction); currentBalance -= amount; // Clear inputs document.getElementById('expenseAmount').value = ''; document.getElementById('expenseDescription').value = ''; updateAllDisplays(); saveData(); showNotification(`Added ${amount.toFixed(2)} expense!`, 'expense'); } function deleteTransaction(id) { if (!isAuthenticated) return; const transaction = transactions.find(t => t.id === id); if (transaction) { if (transaction.type === 'income') { currentBalance -= transaction.amount; } else { currentBalance += transaction.amount; } transactions = transactions.filter(t => t.id !== id); updateAllDisplays(); saveData(); showNotification('Transaction deleted!', 'info'); } } // Links functions function addLink() { if (!isAuthenticated) return; const title = document.getElementById('linkTitle').value; const url = document.getElementById('linkUrl').value; const category = document.getElementById('linkCategory').value; if (!title || !url) { showNotification('Please enter both title and URL', 'error'); return; } // Add protocol if missing let formattedUrl = url; if (!url.startsWith('http://') && !url.startsWith('https://')) { formattedUrl = 'https://' + url; } const link = { id: Date.now(), title: title, url: formattedUrl, category: category || 'General', dateAdded: new Date() }; links.unshift(link); // Clear inputs document.getElementById('linkTitle').value = ''; document.getElementById('linkUrl').value = ''; document.getElementById('linkCategory').value = ''; updateLinksDisplay(); updateDashboardStats(); saveData(); showNotification('Link added successfully!', 'success'); } function deleteLink(id) { if (!isAuthenticated) return; links = links.filter(l => l.id !== id); updateLinksDisplay(); updateDashboardStats(); saveData(); showNotification('Link deleted!', 'info'); } // Todo functions function addTodo() { if (!isAuthenticated) return; const title = document.getElementById('todoTitle').value; const description = document.getElementById('todoDescription').value; const priority = document.getElementById('todoPriority').value; const dueDate = document.getElementById('todoDueDate').value; if (!title.trim()) { showNotification('Please enter a task title', 'error'); return; } const todo = { id: Date.now(), title: title.trim(), description: description.trim(), priority: priority, dueDate: dueDate ? new Date(dueDate) : null, completed: false, createdAt: new Date() }; todos.unshift(todo); // Clear inputs document.getElementById('todoTitle').value = ''; document.getElementById('todoDescription').value = ''; document.getElementById('todoDueDate').value = ''; updateTodosDisplay(); updateDashboardStats(); saveData(); showNotification('Task added successfully!', 'success'); } function toggleTodo(id) { if (!isAuthenticated) return; const todo = todos.find(t => t.id === id); if (todo) { todo.completed = !todo.completed; updateTodosDisplay(); updateDashboardStats(); saveData(); showNotification(todo.completed ? 'Task completed!' : 'Task reopened!', 'info'); } } function deleteTodo(id) { if (!isAuthenticated) return; todos = todos.filter(t => t.id !== id); updateTodosDisplay(); updateDashboardStats(); saveData(); showNotification('Task deleted!', 'info'); } // Image gallery functions function handleDragOver(e) { e.preventDefault(); e.currentTarget.classList.add('dragover'); } function handleDragLeave(e) { e.currentTarget.classList.remove('dragover'); } function handleDrop(e) { e.preventDefault(); e.currentTarget.classList.remove('dragover'); const files = e.dataTransfer.files; handleFiles(files); } function handleFileSelect(e) { const files = e.target.files; handleFiles(files); } function handleFiles(files) { if (!isAuthenticated) return; for (let file of files) { if (file.type.startsWith('image/')) { const reader = new FileReader(); reader.onload = function(e) { addImageToGallery(e.target.result, file.name); }; reader.readAsDataURL(file); } } } function addImageToGallery(imageSrc, fileName) { if (!isAuthenticated) return; const title = document.getElementById('imageTitle').value || fileName; const category = document.getElementById('imageCategory').value || 'General'; const image = { id: Date.now() + Math.random(), src: imageSrc, title: title, category: category, fileName: fileName, uploadDate: new Date() }; images.unshift(image); // Clear inputs document.getElementById('imageTitle').value = ''; document.getElementById('imageCategory').value = ''; updateGalleryDisplay(); updateDashboardStats(); saveData(); showNotification('Image uploaded successfully!', 'success'); } function deleteImage(id) { if (!isAuthenticated) return; images = images.filter(img => img.id !== id); updateGalleryDisplay(); updateDashboardStats(); saveData(); showNotification('Image deleted!', 'info'); } function openImageModal(imageSrc) { document.getElementById('modalImage').src = imageSrc; document.getElementById('imageModal').classList.add('active'); } function closeModal() { document.getElementById('imageModal').classList.remove('active'); } // Display update functions function updateAllDisplays() { updateBalance(); updateRevenueStats(); updateTransactionsList(); updateLinksDisplay(); updateTodosDisplay(); updateGalleryDisplay(); updateDashboardStats(); } function updateBalance() { document.getElementById('currentBalance').textContent = `${currentBalance.toFixed(2)}`; document.getElementById('dashboardBalance').textContent = `${currentBalance.toFixed(2)}`; } function updateRevenueStats() { const today = new Date(); const startOfToday = new Date(today.getFullYear(), today.getMonth(), today.getDate()); const startOfWeek = new Date(today.getFullYear(), today.getMonth(), today.getDate() - today.getDay()); const startOfMonth = new Date(today.getFullYear(), today.getMonth(), 1); const todayRevenue = transactions .filter(t => t.type === 'income' && t.date >= startOfToday) .reduce((sum, t) => sum + t.amount, 0); const weekRevenue = transactions .filter(t => t.type === 'income' && t.date >= startOfWeek) .reduce((sum, t) => sum + t.amount, 0); const monthRevenue = transactions .filter(t => t.type === 'income' && t.date >= startOfMonth) .reduce((sum, t) => sum + t.amount, 0); const totalExpenses = transactions .filter(t => t.type === 'expense') .reduce((sum, t) => sum + t.amount, 0); document.getElementById('todayRevenue').textContent = `${todayRevenue.toFixed(2)}`; document.getElementById('weekRevenue').textContent = `${weekRevenue.toFixed(2)}`; document.getElementById('monthRevenue').textContent = `${monthRevenue.toFixed(2)}`; document.getElementById('totalExpenses').textContent = `${totalExpenses.toFixed(2)}`; document.getElementById('dashboardTodayRevenue').textContent = `${todayRevenue.toFixed(2)}`; } function updateTransactionsList() { const listElement = document.getElementById('transactionList'); if (transactions.length === 0) { listElement.innerHTML = `
No transactions yet. Add some revenue or expenses to get started!
`; return; } const recentTransactions = transactions.slice(0, 10); listElement.innerHTML = recentTransactions.map(transaction => `
${transaction.description}
${transaction.category}
${transaction.date.toLocaleDateString()} ${transaction.date.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}
${transaction.type === 'income' ? '+' : '-'}${transaction.amount.toFixed(2)}
`).join(''); } function updateLinksDisplay() { const gridElement = document.getElementById('linksGrid'); if (links.length === 0) { gridElement.innerHTML = `
No links added yet. Add your favorite websites above!
`; return; } gridElement.innerHTML = links.map(link => `
${link.title}
${link.url}
${link.category}
`).join(''); } function updateTodosDisplay() { const todosElement = document.getElementById('todosList'); if (todos.length === 0) { todosElement.innerHTML = `
No tasks yet. Add some tasks to get started!
`; return; } // Sort todos: incomplete first, then by priority const sortedTodos = [...todos].sort((a, b) => { if (a.completed !== b.completed) { return a.completed - b.completed; } const priorityOrder = { high: 3, medium: 2, low: 1 }; return priorityOrder[b.priority] - priorityOrder[a.priority]; }); todosElement.innerHTML = sortedTodos.map(todo => `
${todo.title}
${todo.description ? `
${todo.description}
` : ''}
${todo.priority.toUpperCase()} ${todo.dueDate ? `Due: ${todo.dueDate.toLocaleDateString()}` : ''}
`).join(''); } function updateGalleryDisplay() { const galleryElement = document.getElementById('galleryGrid'); if (images.length === 0) { galleryElement.innerHTML = `
No images yet. Upload some images to get started!
`; return; } galleryElement.innerHTML = images.map(image => `
${image.title}
${image.category}
`).join(''); } function updateDashboardStats() { const activeTodos = todos.filter(t => !t.completed).length; document.getElementById('dashboardTotalTodos').textContent = activeTodos; document.getElementById('dashboardTotalLinks').textContent = links.length; document.getElementById('dashboardTotalImages').textContent = images.length; } function showNotification(message, type) { const notification = document.createElement('div'); const colors = { success: '#27ae60', error: '#e74c3c', expense: '#e74c3c', info: '#3498db' }; notification.style.cssText = ` position: fixed; top: 20px; right: 20px; padding: 15px 25px; background: ${colors[type] || colors.info}; color: white; border-radius: 10px; box-shadow: 0 4px 15px rgba(0,0,0,0.2); z-index: 1000; transform: translateX(400px); transition: transform 0.3s ease; font-weight: 500; `; notification.textContent = message; document.body.appendChild(notification); setTimeout(() => notification.style.transform = 'translateX(0)', 100); setTimeout(() => notification.style.transform = 'translateX(400px)', 3000); setTimeout(() => document.body.removeChild(notification), 3300); } // Initialize the app init(); All-in-One Dashboard
